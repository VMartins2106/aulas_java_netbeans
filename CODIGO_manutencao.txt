package pct;

import java.util.Scanner;

/**
 *
 * @author manager
 */
public class Manutencao_ConversorTemp_Nicolas {
    
    public static void main(String[] args) {
        
        //Declaração de Variaveis
        double cel, fah;
        int op;
        
        //Criação e Instancia do Objeto de Entrada
        Scanner entrada = new Scanner(System.in);
        
        
        do{
            
        //apresentação
        System.out.println("\n\t\t -- Conversor de Temperatura -- \n");
        
        //Menu
        
        System.out.println("[1]. Converter de ºC para ºF :");
        System.out.println("[2]. Converter de ºF para ºC :");
        System.out.println("[3]. Sair do conversor.");
        
        //entrada
        
        System.out.print("\n\tOpção: ");
        op = entrada.nextInt();
        
        if (op == 1) {
            
        System.out.print("Informe a Temperatura (C°): ");
        cel = entrada.nextInt();

        //Processamento
        fah = 9 * cel / 5 + 32;

        //saida
        System.out.println("\n\t F° = "+ fah +"\t\n ");

       }
        else if (op == 2){
            
        System.out.println("Informe a Temperatura (ºF) :");
        fah = entrada.nextInt();
        
        //Processamento
        
        cel = (fah - 32)/1.8;
        
        //Saída
        System.out.println("\n\t C° = "+ cel +"\t\n ");
        }
        
        else if (op == 3){
            
        System.out.println("\n\tVocê escolheu sair do conversor!");
            
            }
        }while (op != 3);
    } 
}
